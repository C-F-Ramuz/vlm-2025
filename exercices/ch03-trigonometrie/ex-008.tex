\begin{itemize}
	\item[$\bullet$]
	$\triangle\mbox{ AEF rect. en E }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\sin{(15\degree)}\;=\;\dfrac{AE}{10}$\\
	$ $\\
	$\Longrightarrow AE\;=\;\sin{(15\degree)}\cdot10\;\cong\;2,588\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  EB\;=\;\sqrt{10^2-2,588^2}\;\cong\;9,659\mbox{ cm}$\\
	$ $\\
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\item[$\bullet$]
	$\triangle\mbox{ ADB rect. en D }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\widehat{ABD}\;=\;15+20\;=\;35\degree$ car adjacents.\\
	$ $\\
	$\cos{(35\degree)}\;=\;\dfrac{BD}{10}$\\
	$ $\\
	$\Longrightarrow BD\;=\;\cos{(35\degree)}\cdot10\;\cong\;8,192\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  AD\;=\;\sqrt{10^2-8,192^2}\;\cong\;5,736\mbox{ cm}$\\
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\newpage
	\item[$\bullet$]
	$\triangle\mbox{ ABC rect. en B }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\sin{(60\degree)}\;=\;\dfrac{10}{AC}$\\
	$ $\\
	$\Longrightarrow AC\;=\;\dfrac{10}{\sin{(15\degree)}}\;\cong\;11,547\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  BC\;=\;\sqrt{11,547^2-10^2}\;\cong\;5,774\mbox{ cm}$\\
	$ $\\	
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\item[$\bullet$]
	$\triangle\mbox{ AEG rect. en E }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\widehat{BAC}\;=\;180-90-60\;=\;30\degree$ car $\sum \sphericalangle \triangle \;=\; 180\degree$\\
	$ $\\
	$\widehat{BGA}\;=\;180-15-30\;=\;135\degree $ car $\sum \sphericalangle \triangle \;=\; 180\degree$\\
	$ $\\
	$\widehat{AGF}\;=\;180-135\;=\;45\degree$ car supplémentaires\\
	$ $\\
	$\tan{(45\degree)}\;=\;\dfrac{2,588}{GE}$\\
	$ $\\
	$\Longrightarrow GE\;=\;\dfrac{2,588}{\tan{(45\degree)}}\;\cong\;2,588\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  AG\;=\;\sqrt{2,588^2+2,588^2}\;\cong\;3,660\mbox{ cm}$\\
	$ $\\	
	$\Longrightarrow BG\;=\;BE-GE\;=\;9,659-2,588\;\cong\;5,681\mbox{ cm}$\\
	$ $\\
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\item[$\bullet$]
	$\triangle\mbox{ ADH rect. en D }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\widehat{DAB}\;=\;180-90-35\;=\;55\degree$ car $\sum \sphericalangle \triangle \;=\; 180\degree$\\
	$ $\\
	$\widehat{DAH}\;=\;55-30\;=\;25\degree$ car adjacents\\
	$ $\\
	$\tan{(25\degree)}\;=\;\dfrac{HD}{5,736}$\\
	$ $\\
	$\Longrightarrow HD\;=\;\tan{(25\degree)}\cdot5,736\;\cong\;2,692\mbox{ cm}$\\
	$ $\\
	$\Longrightarrow BH\;=\;BD-HD\;=\;8,192-2,692\;\cong\;5,499\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  AH\;=\;\sqrt{5,736^2+2,692^2}\;\cong\;6,370\mbox{ cm}$\\
	$ $\\
	$\Longrightarrow CH\;=\;AC-AH\;=\;11,547-6,370\;\cong\;5,177\mbox{ cm}$\\
	%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
	\newpage
	\item[$\bullet$]
	$\triangle\mbox{ AEF rect. en E }\Rightarrow\mbox{ trigo + Pyth }$\\
	$ $\\
	$\widehat{EAB}\;=\;180-90-15\;=\;75\degree$ car $\sum \sphericalangle \triangle \;=\; 180\degree$\\
	$ $\\
	$\widehat{EAF}=75-25-30=20\degree$ car adjacents\\
	$ $\\
	$\cos{(20\degree)}\;=\;\dfrac{2,588}{AF}$\\
	$ $\\
	$\Longrightarrow AF\;=\;\dfrac{2,588}{\cos{(20\degree)}}\;\cong\;2,754\mbox{ cm}$\\
	$ $\\
	$\mbox{(Pyth) }  EF\;=\;\sqrt{2,754^2-2,588^2}\;\cong\;0,942\mbox{ cm}$\\
	$ $\\
	$\Longrightarrow FG\;=\;EG-GF\;=\;2,588-0,942\;\cong\;1,646\mbox{ cm}$\\
	$ $\\
	$\Longrightarrow DF\;=\;AD-AF\;=\;5,736-2,754\;\cong\;3,019\mbox{ cm}$
\end{itemize}

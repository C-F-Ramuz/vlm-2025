\chapter{Algèbre}

\section{Équations}
\begin{exercice}{EX-CH01-001}{Résolution d'équation du second degré}

Résoudre dans $\mathbb{R}$ les équations suivantes~:

\begin{enumerate}[label=\alph*)]
  \item $2x^2 - 5x + 3 = 0$
  \item $x^2 + 4x + 5 = 0$
  \item $3x^2 - 12 = 0$
\end{enumerate}

\end{exercice}
\begin{solution}{EX-CH01-001}

\begin{enumerate}[label=\alph*)]

  \item $2x^2 - 5x + 3 = 0$

  $\Delta = (-5)^2 - 4 \cdot 2 \cdot 3 = 25 - 24 = 1 > 0$

  Deux solutions~: $x_1 = \dfrac{5+1}{4} = \dfrac{3}{2}$ \quad et \quad
  $x_2 = \dfrac{5-1}{4} = 1$

  \item $x^2 + 4x + 5 = 0$

  $\Delta = 16 - 20 = -4 < 0$ \quad Pas de solution réelle.

  \item $3x^2 - 12 = 0 \;\Rightarrow\; x^2 = 4$

  $x_1 = 2$ \quad et \quad $x_2 = -2$

\end{enumerate}

\end{solution}

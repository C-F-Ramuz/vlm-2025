\chapter{Trigonométrie}

\section{Cercle trigonométrique}
\begin{exercice}{EX-CH03-007}{Valeurs exactes du cercle trigonométrique}

Sans calculatrice, donner les valeurs exactes de~:

\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
  \item $\sin\!\left(\dfrac{\pi}{6}\right)$
  \item $\cos\!\left(\dfrac{\pi}{4}\right)$
  \item $\tan\!\left(\dfrac{\pi}{3}\right)$
  \item $\sin\!\left(\dfrac{3\pi}{2}\right)$
  \item $\cos(\pi)$
  \item $\tan\!\left(\dfrac{5\pi}{4}\right)$
\end{enumerate}
\end{multicols}

\end{exercice}
\begin{solution}{EX-CH03-007}

\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
  \item $\sin\!\left(\dfrac{\pi}{6}\right) = \dfrac{1}{2}$
  \item $\cos\!\left(\dfrac{\pi}{4}\right) = \dfrac{\sqrt{2}}{2}$
  \item $\tan\!\left(\dfrac{\pi}{3}\right) = \sqrt{3}$
  \item $\sin\!\left(\dfrac{3\pi}{2}\right) = -1$
  \item $\cos(\pi) = -1$
  \item $\tan\!\left(\dfrac{5\pi}{4}\right) = 1$
\end{enumerate}
\end{multicols}

\end{solution}
